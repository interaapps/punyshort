Examples




Python


link = "https://interaapps.de"

import urllib.request, json
with urllib.request.urlopen("https://www.punyshort.ga/api/jsonapi.php?get_informations="+link) as url:
    data = json.loads(url.read().decode())
print(data)
print("Clicks"+data["clicks"])
print("New: "+data["new"])



JavaScript


function loadJSON(path, success, error)
{
var xhr = new XMLHttpRequest();
xhr.onreadystatechange = function()
{
if (xhr.readyState === XMLHttpRequest.DONE) {
if (xhr.status === 200) {
if (success)
success(JSON.parse(xhr.responseText));
} else {
if (error)
error(xhr);
}
}
};

xhr.open("GET", path, true);
xhr.send();
}

var obj;

loadJSON('https://www.punyshort.ga/api/jsonapi.php?shortlink=https://interaapps.de',
function(data) {
obj = data;
console.log("link: pnsh.ga/" + obj['url']);
console.log("Clicks: " + obj['clicks']);
console.log("New: " + obj['new']);
},
function(xhr) { console.error(xhr); },

);
